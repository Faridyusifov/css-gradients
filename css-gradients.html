CSS Gradients

Linear-gradient

.gradient{
      background-image : Linear-gradient ( green, orange ) ;
}
.gradient{
      background-image : Linear-gradient (to left, green, blue ) ;
}
.gradient{
      background-image : Linear-gradient (to top left, red, orange ) ;
}
.gradient{
      background-image : Linear-gradient (50deg, gold, purple ) ;
}
.gradient{
      background-image : Linear-gradient (to right, black , darkblue 50% ) ;
}

Radial Gradients

.gradient{
      background-image : radial-gradient ( yellow, black ) ;
}
.gradient{
      background-image : radial-gradient ( circle ,purple, black ) ;
}
.gradient{
      background-image : radial-gradient ( circle closest-side ,yellow, black ) ;
}
.gradient{
      background-image : radial-gradient ( circle at top right ,yellow, red ) ;
}

Conic Gradients

.conic-gradient{
